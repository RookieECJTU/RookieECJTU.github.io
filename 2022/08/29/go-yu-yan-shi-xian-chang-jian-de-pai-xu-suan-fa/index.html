Go语言实现常见的排序算法

Go 算法
发布日期:   2022-08-29

排序算法

排序算法 平均时间复杂度 最好情况 最坏情况 空间复杂度 稳定性
冒泡排序 O(n²) O(n) O(n²) O(1) 稳定
选择排序 O(n²) O(n²) O(n²) O(1) 不稳定
插入排序 O(n²) O(n) O(n²) O(1) 稳定
希尔排序 O(n㏒n) O(n㏒²n) O(n㏒²n) O(1) 不稳定
归并排序 O(n㏒n) O(n㏒n) O(n㏒n) O(n) 稳定
快速排序 O(n㏒n) O(n㏒n) O(n²) O(㏒n) 不稳定
堆排序 O(n㏒n) O(n㏒n) O(n㏒n) O(1) 不稳定
计数排序 O(n+k) O(n+k) O(n+k) O(k) 稳定

  • 稳定：如果a原本在b前面，而a=b，排序之后a仍然在b的前面；

  • 不稳定：如果a原本在b的前面，而a=b，排序之后a可能会出现在b的后面；

  • 时间复杂度: 一个算法执行所耗费的时间。

  • 空间复杂度: 运行完一个程序所需内存的大小。

一、 冒泡排序

// 冒泡排序
// 相邻两位交换，把最大数移动到最右边，遍历n次序列。
func bubbleSort(arr []int) {
	length := len(arr)
	for i := 0; i < length; i++ {
		// 判断序列是否已经有序的标志位，避免无效循环
		flag := true
		// 每轮遍历会把最大值移动到右边
		for j := 0; j < length-i; j++ {
			if arr[j] > arr[j+1] {
				arr[j], arr[j+1] = arr[j+1], arr[j]
				flag = false
			}
		}
		// true表示序列已经有序，不用继续遍历了
		// false表示本次遍历发生了交换，需要继续判断下个循环
		if flag {
			break
		}
	}
}

二、 选择排序

// 选择排序
// 遍历序列找到最小值放最左边，循环遍历剩下的序列。
func selectSort(arr []int) {
	length := len(arr)
	for i := 0; i < length; i++ {
		min := i
		for j := i+1; j < length; j++ {
			// 如果下标j的值比min小，更新下标min
			if arr[min] > arr[j] {
				min = j
			}
		}
		// 遍历一轮序列，最小值发生变化
		if min != i {
			// 交换下标为i,min的值
			arr[i], arr[min] = arr[min], arr[i]
		}
	}
}

三、 插入排序

// 插入排序
// 取首元素作为有序队列，把第二位插入到有序队列中，把第三位插入到前两位组成的有序队列中
// 新的一位插入有序队列时，跟他的前一位比较，即有序队列的最右边，依次向前遍历。
func insertSort(arr []int) {
	length := len(arr)
	for i := 1; i < length; i++ {
		// 大于等于有序队列中最大元素，直接插入队尾
		if arr[i] >= arr[i-1] {
			continue
		}
		temp := arr[i]
		j := i-1
		//temp < arr[j]，把j向后移位，依次循环，直到temp >= arr[j]
		for ; j >= 0 && temp < arr[j]; j-- {
			arr[j+1] = arr[j]
		}
		// 发生了j--，需要j+1
		arr[j+1] = temp
	}
}

四、 希尔排序

希尔排序，也称递减增量排序算法，是插入排序的一种更高效的改进版本。但希尔排序是非稳定排序算法。

希尔排序是基于插入排序的以下两点性质而提出改进方法的：

  • 插入排序在对几乎已经排好序的数据操作时，效率高，即可以达到线性排序的效率；
  • 但插入排序一般来说是低效的，因为插入排序每次只能将数据移动一位；

希尔排序的基本思想是：先将整个待排序的记录序列分割成为若干子序列分别进行直接插入排序，待整个序列中的记录”基本有序”时，再对全体记录进行依次直接插入排序。

// 希尔排序
// 按间隔分组，每组进行插入排序。间隔逐渐减小，直到间隔为1
func shellSort(arr []int) {
	length := len(arr)
	for gap := length / 2; gap > 0; gap /= 2 {
		for i := gap; i < length; i++ {
			if arr[i] >= arr[i-gap] {
				continue
			}
			temp := arr[i]
			j := i - gap
			for ; j >= 0 && temp < arr[j]; j -= gap {
				arr[j+gap] = arr[j]
			}
			arr[j+gap] = temp
		}
	}
}

五、 归并排序

// 归并排序
// 将长度为n的序列划分成n个长度为1的子序列，两两归并成n/2个长度为2子序列，
// 不断两两归并得到1个长度为n的子序列
func mergeSort(arr []int) []int {
	length := len(arr)
	if length <= 1 {
		return arr
	}
	num := length/2
	left := mergeSort(arr[:num])
	right := mergeSort(arr[num:])
	return merge(left, right)
}

func merge(left, right []int) []int {
	l, r := 0, 0
	res := make([]int,0)
	// left或right有一方遍历完则退出循环
	for l < len(left) && r < len(right) {
		if left[l] < right[r] {
			res = append(res, left[l])
			l++
		} else {
			res = append(res, right[r])
			r++
		}
	}
	// left和right均为有序，直接将剩余部分加进序列
	// 如果上面是left遍历完，left[l:]为[]，right还有剩余值
	// 如果上面是right遍历完，right[r:]为[], left还有剩余值
	res = append(res, left[l:]...)
	res = append(res, right[r:]...)
	return res
}

六、 快速排序

递归版

// 快速排序(递归版)
// 取首位元素为基准，遍历序列将小于基准的数字放左边，大于基准的数字放右边。递归处理基准左右两个序列
func quickSort(arr []int) {
	length := len(arr)
	if length <= 1 {
		return
	}
	quick(arr, 0, length-1)
}

func quick(arr []int, start, end int) {
	if start >= end {
		return
	}
	// 把第一个元素作为基准
	pivot := arr[start]
	i,j := start, end
	for i < j {
		// 从右往左找小于基准的数字，大于则不处理。不取等会死循环
		for i < j && arr[j] >= pivot {
			j--
		}
		// 从左往右找大于基准的数字，小于则不处理
		for i < j && arr[i] <= pivot {
			i++
		}
		// 如果存在大于基准的数或者小于基准的，移动到i位置
		arr[i], arr[j] = arr[j], arr[i]
	}
	// 把基准放在i,j重合位置
	arr[i],arr[start] = pivot,arr[i]
	// 递归处理基准左边的序列
	quick(arr, start, i-1)
	// 递归处理基准右边的序列
	quick(arr, i+1, end)
}

非递归版

// 快速排序(非递归版)
// 需要借助额外空间记录需要快排的序列起点和终点。
func quickSort(arr []int) {
	// 切片模拟栈
	stack := make([]int,0)
	start, end := 0, len(arr)-1
	// 一轮快排，将序列中小于基准的移动到左边，大于基准的移动到右边
	mid := quick1(arr, start, end)
	// 长度大于2的子序列才需要继续快排处理
	if start < mid-1 {
		push(&stack, start)
		push(&stack, mid-1)
	}
	if end > mid+1 {
		push(&stack, mid+1)
		push(&stack, end)
	}
	// 遍历栈，直到所有需要快排的序列全部排序完
	for len(stack) > 0 {
		end = pop(&stack)
		start = pop(&stack)
		mid = quick1(arr, start, end)
		// 长度大于2的子序列才需要继续快排处理
		if start < mid-1 {
			push(&stack, start)
			push(&stack, mid-1)
		}
		if end > mid+1 {
			push(&stack, mid+1)
			push(&stack, end)
		}
	}
}

// 模拟栈的push
func push(stack *[]int, num int) {
	*stack = append(*stack, num)
}

// 模拟栈的pop
func pop(stack *[]int) int {
	length := len(*stack)
	if length == 0 {
		return 0
	}
	res := (*stack)[length-1]
	*stack = (*stack)[:length-1]
	return res
}

// 快排处理，将序列中小于基准的移动到左边，大于基准的移动到右边
func quick(arr []int, start, end int) int {
	// 把第一个元素作为基准
	pivot := arr[start]
	i,j := start, end
	for i < j {
		// 从右往左找小于基准的数字，大于则不处理。不取等会死循环
		for i < j && arr[j] >= pivot {
			j--
		}
		// 从左往右找大于基准的数字，小于则不处理
		for i < j && arr[i] <= pivot {
			i++
		}
		// 如果存在大于基准的数或者小于基准的，移动到i位置
		arr[i], arr[j] = arr[j], arr[i]
	}
	// 把基准放在i,j重合位置
	arr[i],arr[start] = pivot,arr[i]
	return i
}

快速排序可优化点

  • 时间复杂度优化，基准选择不恰当会导致时间复杂度退化,极限情况下相当于冒泡排序，即复杂度为O(n^2)。可以取多个数字，将中位数作为基准
  • 递归深度优化，快排会将序列拆分成大量有一定顺序的小区间，此时可以使用插入排序。

七、堆排序

// 堆排序
// 升序使用大顶堆，降序使用小顶堆
// 根据序列进行建堆，将堆顶元素和末位元素交换，去掉末位，重新调整剩下序列的大顶堆
func heapSort(arr []int) []int {
	length := len(arr)
	// 调整序列为大顶堆，最后一个非叶子节点(length-1-1)/2
	for i := length/2 - 1; i >= 0; i-- {
		heapify(arr, i, length)
	}
	for i := length - 1; i > 0; i-- {
		// 首尾元素交换
		arr[0], arr[i] = arr[i], arr[0]
		// 去掉最大元素，剩下的序列重新调整出大顶堆
		heapify(arr, 0, i)
	}
	return arr
}

// 调整二叉树为大顶堆
func heapify(arr []int, i, length int) {
	// 默认i最大
	largest := i
	// 非叶子节点i的左右孩子
	left := 2*i+1
	right := 2*i+2
	//存在左孩子且左孩子大，最大指向left
	if left < length && arr[left] > arr[largest] {
		largest = left
	}
	//存在右孩子且右孩子更大，最大指向right
	if right < length && arr[right] > arr[largest] {
		largest = right
	}
	//最大发生过改变，交换
	if largest != i {
		arr[i], arr[largest] = arr[largest], arr[i]
		// 递归维护下一层，直到最后一层非叶子节点
		heapify(arr,largest, length)
	}
}

八、计数排序

// 计数排序
// 需要一个额外空间记录每个元素出现的次数，然乎从小到大把元素放回原数组中
func countSort(arr []int) {
	min, max := 0,0
	length := len(arr)
	// 遍历序列获取最大值和最小，来确认需要申请的空间大小
	for i := 0; i < length; i++ {
		if min > arr[i] {
			min = arr[i]
		}
		if max < arr[i] {
			max = arr[i]
		}
	}
	// 创建计数器
	count := max - min + 1
	bucket := make([]int, count)
	// 统计每个元素的个数
	for i := 0; i < length; i++ {
		bucket[arr[i]-min]++
	}
	// 按元素从小到达放回原数组
	index := 0
	for i := 0; i < len(bucket); i++ {
		for j := 0; j < bucket[i]; j++ {
			arr[index] = i + min
			index++
		}
	}
}
